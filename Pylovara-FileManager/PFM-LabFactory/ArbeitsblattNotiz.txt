#!/bin/bash
# bootstrap.sh
REPO="https://github.com/Pylovara/PFM-LaboFactory"
DEST="$HOME"

echo ">> Klone Grundstruktur von $REPO nach $DEST"
git clone --depth=1 "$REPO" "$DEST/PFM-LaboFactory"

echo ">> Richte Skelett ein..."
cd "$DEST/PFM-LaboFactory/SKELETT" && cp -r . "$HOME"

echo ">> Skelett erfolgreich gesetzt. Starte Pylovara-FM..."
bash ~/.config/hypr/Hyprland-Module/Pylovara-FileManager/start.sh
